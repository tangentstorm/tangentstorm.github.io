2 part 2 : the escape

2.1 objects from records

type
  pDat = ^tDat;
  tDat = array of byte;

  pStr = ^tStr;
  tStr = array of char;

  pTyp = ^tTyp;
  tTyp = record
    sup : pTyp;
    sym : pStr;
  end;

  tVal = record
    typ : pTyp;
    dat : pDat;
  end;

  { the four fundamental messaging patterns }
  tVoMsg = procedure( obj : pObj );
  tRuMsg = procedure( obj : pObj; msg : pMsg );
  tNaMsg = funtion( obj : pObj ) : pMsg;
  tTiMsg = funtion( obj : pObj; msg : pMsg ) : pMsg;

  pObj = ^tObj;
  tObj = record
    val : tVAL;
     vo : tVO;
     ru : tRU;
     na : tNA;
     ti : tTI;
  end;
