  ;; I'd call 'val 'row except '(first row) etc. looks like they're
  ;; talking about rows when really they're talking about values. :/
  (progn
    (if (eq val 'hline)
        (progn (add-to-list 'combined-keyword-set 
                            (cons 'sequence current-keyword-set))
               (set 'current-keyword-set initial-keyword-set))
      ;; else:
      (when (not (equal (first val) "!"))
        (let* ((word  (upcase (second val)))
               (fg (either (third val) "white"))
               (bg (either (fourth val)  "cornflower blue"))
               (face (intern (concat "sabren-org-keyword-" (downcase word)))))

                     (setq current-keyword-set
                           (remove "DONE" current-keyword-set)))

            ;; treat everything else as keywords
            (progn (add-to-list 'current-keyword-set word)
                   (princ (pp-to-string `(defface-mjw ,word ,face ,fg ,bg))))))))))

(defun upcase-pascal-words ()
  ;; this is a little more aggressive than the one that comes with pascal.el
  (interactive)
  (mapcar
   (lambda (sym)
     (save-excursion
       (beginning-of-buffer)
       (replace-regexp (token-match-symbol sym)
                       (upcase (symbol-name sym)))))

   ;; special turbo pascal keywords:
   '(asm unit uses implementation interface
         assembler virtual absolute virtual

         ;; constants and operators
         nil true false and or not xor mod div

         ;; keywords
         type var const record array object 
         procedure function constructor destructor
         begin end if then while do repeat until for to do case else elsif

         ;; types
         byte shortint integer longint word 
         char string boolean

         ;; builtin functions
         close blockread ioresult reset
         length sizeof new free

         ;; flow control
         exit
         )))

(defun insert-balanced-comments (arg)
  "Insert a set of balanced comments around the s-expression 
containing the point.  If this command is invoked repeatedly
(without any other command occurring between invocations), the 
comment progressively moves outward over enclosing expressions."
  (interactive "*p")
  (save-excursion
    (when (eq last-command this-command)
      (when (search-backward "#|" nil t)
        (save-excursion
          (delete-char 2)
          (while (and (< (point) (point-max)) (not (looking-at " *|#")))
            (forward-sexp))
          (replace-match ""))))
    (while (> arg 0)
      (backward-char 1)
      (cond ((looking-at ")") (incf arg))
            ((looking-at "(") (decf arg))))
    (insert "#|")
    (forward-sexp)
    (insert "|#")))

      (delete-char 2)
      (while (and (< (point) (point-max)) (not (looking-at " *|#")))

(defun count-words-region (beginning end)
  "Print number of words in the region."
  (interactive "r")
  (message "Counting words in region ... ")

  ;;; 1. Set up appropriate conditions.
  (save-excursion
    (let ((count 0))
      (goto-char beginning)

      ;;; 2. Run the while loop.
      (while (and (< (point) end)
                  (re-search-forward "\\w+\\W*" end t))
        (setq count (1+ count)))

      ;;; 3. Send a message to the user.
      (cond ((zerop count)
             (message
              "The region does NOT have any words."))
            ((= 1 count)
             (message
              "The region has 1 word."))
            (t
             (message
              "The region has %d words." count))))))

(global-set-key "\C-cw" 'count-words-region)
